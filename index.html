Notification.Builder notification=new Notification.Builder(this);

PendingIntent.FLAG_CANCEL_CURRENT);
Notification n=notification.setContentIntent(intent).setSmallIcon(R.mipmap.ic_launcher)
.setContentTitle(“aa”).setContentText(“bbb”).build();
n.sound= Uri.parse(“/system/media/audio/notifications/Heaven.ogg”);
long[] vibrates= {0, 1000, 1000, 1000};
n.vibrate=vibrates;
n.ledARGB= 0xFF0000;
n.ledOnMS=1000;
n.ledOffMS=1000;
n.flags=Notification.FLAG_SHOW_LIGHTS;
manager.notify(1,n);
